# .github/workflows/continuous-testing.yml
name: Continuous Testing Pipeline

on:
  push:
    branches: [ main, develop ]
  pull_request:
    branches: [ main ]

jobs:
  # 1. БЫСТРЫЕ ТЕСТЫ (каждый PR)
  fast-tests:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Run Unit Tests
        run: pytest tests/unit -v
      - name: Run Linting
        run: flake8 src/
      - name: Check Coverage
        run: pytest --cov=src --cov-fail-under=80

  # 2. ИНТЕГРАЦИОННЫЕ ТЕСТЫ (push в develop)
  integration-tests:
    runs-on: ubuntu-latest
    needs: fast-tests
    if: github.ref == 'refs/heads/develop'
    steps:
      - uses: actions/checkout@v3
      - name: Start Test Environment
        run: docker-compose -f docker-compose.test.yml up -d
      - name: Run Integration Tests
        run: pytest tests/integration -v
      - name: Stop Test Environment
        run: docker-compose -f docker-compose.test.yml down

  # 3. E2E ТЕСТЫ (push в main)
  e2e-tests:
    runs-on: ubuntu-latest
    needs: integration-tests
    if: github.ref == 'refs/heads/main'
    steps:
      - uses: actions/checkout@v3
      - name: Deploy to Staging
        run: ./deploy-staging.sh
      - name: Run E2E Tests
        run: pytest tests/e2e -v --browser=chrome
      - name: Generate Report
        run: pytest --html=report.html

  # 4. НАГРУЗОЧНЫЕ ТЕСТЫ (ночная сборка)
  performance-tests:
    runs-on: ubuntu-latest
    if: github.event_name == 'schedule'
    steps:
      - uses: actions/checkout@v3
      - name: Run Load Tests
        run: locust -f tests/performance/locustfile.py --headless -u 100 -r 10

# testing_dashboard.py
import json
from datetime import datetime, timedelta
from typing import Dict, List


class ScrumTestingDashboard:
    """Дашборд метрик тестирования для Scrum."""
    
    def __init__(self):
        self.sprint_data = {
            "sprint_number": 5,
            "start_date": "2024-01-01",
            "end_date": "2024-01-14",
            "total_test_cases": 150,
            "executed": 145,
            "passed": 138,
            "failed": 7,
            "blocked": 0,
            "open_bugs": {
                "critical": 0,
                "high": 2,
                "medium": 5,
                "low": 3
            },
            "code_coverage": 85.5,
            "automation_rate": 78.0
        }
        
        self.sprint_history = []
    
    def calculate_sprint_metrics(self) -> Dict:
        """Расчёт метрик текущего спринта."""
        data = self.sprint_data
        
        pass_rate = (data["passed"] / data["executed"]) * 100 if data["executed"] > 0 else 0
        execution_rate = (data["executed"] / data["total_test_cases"]) * 100
        quality_score = pass_rate * (data["code_coverage"] / 100)
        
        return {
            "pass_rate": pass_rate,
            "execution_rate": execution_rate,
            "quality_score": quality_score,
            "bug_density": sum(data["open_bugs"].values()),
            "sprint_health": "Green" if pass_rate >= 95 and quality_score >= 80 else "Red"
        }
    
    def generate_sprint_report(self) -> str:
        """Генерация отчёта по спринту."""
        metrics = self.calculate_sprint_metrics()
        
        return f"""
        ========================================
        SPRINT {self.sprint_data['sprint_number']} TESTING REPORT
        ========================================
        
        📊 Execution Summary
        ───────────────────────────────────────
        Test Cases:      {self.sprint_data['executed']}/{self.sprint_data['total_test_cases']}
        Pass Rate:       {metrics['pass_rate']:.1f}%
        Execution Rate:  {metrics['execution_rate']:.1f}%
        Code Coverage:   {self.sprint_data['code_coverage']:.1f}%
        Automation Rate: {self.sprint_data['automation_rate']:.1f}%
        
        🐛 Open Bugs
        ───────────────────────────────────────
        Critical: {self.sprint_data['open_bugs']['critical']}
        High:     {self.sprint_data['open_bugs']['high']}
        Medium:   {self.sprint_data['open_bugs']['medium']}
        Low:      {self.sprint_data['open_bugs']['low']}
        
        📈 Quality Metrics
        ───────────────────────────────────────
        Quality Score:   {metrics['quality_score']:.1f}
        Sprint Health:   {metrics['sprint_health']}
        
        ✅ Recommended Actions
        ───────────────────────────────────────
        {self._generate_recommendations(metrics)}
        """
    
    def _generate_recommendations(self, metrics: Dict) -> str:
        """Генерация рекомендаций."""
        recommendations = []
        
        if metrics['pass_rate'] < 90:
            recommendations.append("- Increase focus on fixing failing tests")
        if metrics['execution_rate'] < 95:
            recommendations.append("- Complete remaining test execution")
        if self.sprint_data['open_bugs']['critical'] > 0:
            recommendations.append("- CRITICAL: Resolve critical bugs before release")
        
        return "\n".join(recommendations) if recommendations else "- All metrics are healthy. Keep up the good work!"


# Пример использования
dashboard = ScrumTestingDashboard()
print(dashboard.generate_sprint_report())
