### 2.2 Классификация дефектов

| Критерий | Значения | Описание |
|:---|:---|:---|
| **Серьёзность (Severity)** | Critical | Критический — система не работает |
| | High | Высокий — ключевая функция не работает |
| | Medium | Средний — функция работает с ограничениями |
| | Low | Низкий — косметическая проблема |
| **Приоритет (Priority)** | High | Высокий — исправить срочно |
| | Medium | Средний — исправить в ближайший релиз |
| | Low | Низкий — исправить при возможности |
| **Тип (Type)** | Functional | Функциональный |
| | UI/UX | Интерфейсный |
| | Performance | Производительности |
| | Security | Безопасности |

3.2 JQL (Jira Query Language)
sql
-- Поиск дефектов
project = "SHOP" AND issuetype = Bug AND status = Open

-- Дефекты за последнюю неделю
created >= -7d AND issuetype = Bug

-- Дефекты с высоким приоритетом
priority = High AND status != Closed

-- Мои открытые задачи
assignee = currentUser() AND status != Closed

-- Дефекты по компоненту
component = "Payment" AND issuetype = Bug

-- Дефекты, требующие внимания
priority in (Critical, High) AND status in (Open, "In Progress")

4.3 Интеграция Jira и TestRail
python
# Пример синхронизации Jira и TestRail через API
import requests

class JiraTestRailSync:
    def __init__(self, jira_url, jira_token, testrail_url, testrail_api_key):
        self.jira_url = jira_url
        self.jira_token = jira_token
        self.testrail_url = testrail_url
        self.testrail_api_key = testrail_api_key
    
    def create_bug_from_failed_test(self, test_result):
        """
        Создаёт баг в Jira из упавшего тест-кейса.
        """
        bug_data = {
            "fields": {
                "project": {"key": "TEST"},
                "summary": f"[Auto] {test_result['test_title']} failed",
                "description": self._format_description(test_result),
                "issuetype": {"name": "Bug"},
                "priority": {"name": "High"}
            }
        }
        
        response = requests.post(
            f"{self.jira_url}/rest/api/2/issue",
            headers={"Authorization": f"Bearer {self.jira_token}"},
            json=bug_data
        )
        
        return response.json()
    
    def _format_description(self, test_result):
        return f"""
h3. Неудачный тест
Тест: {test_result['test_title']}
ID: {test_result['test_id']}

h3. Шаги воспроизведения
{test_result['steps']}

h3. Фактический результат
{test_result['actual_result']}

h3. Ожидаемый результат
{test_result['expected_result']}
"""
5. Метрики для тест-менеджмента
5.1 Основные метрики
Метрика	Формула	Назначение
Test Pass Rate	(Пройдено / Всего) × 100%	Процент успешных тестов
Test Execution Rate	(Выполнено / Запланировано) × 100%	Выполнение плана
Defect Density	Дефекты / KLOC	Качество кода
Defect Leakage	(Дефекты в продакшене / Всего дефектов) × 100%	Утечка дефектов
Defect Resolution Time	Время от открытия до закрытия	Эффективность команды
5.2 Дашборд тестирования
python
class TestingDashboard:
    """Дашборд метрик тестирования."""
    
    def __init__(self):
        self.metrics = {
            "total_test_cases": 0,
            "passed": 0,
            "failed": 0,
            "blocked": 0,
            "not_run": 0,
            "open_bugs": 0,
            "critical_bugs": 0,
            "high_bugs": 0
        }
    
    def pass_rate(self) -> float:
        if self.metrics["total_test_cases"] == 0:
            return 0.0
        return (self.metrics["passed"] / self.metrics["total_test_cases"]) * 100
    
    def execution_rate(self) -> float:
        if self.metrics["total_test_cases"] == 0:
            return 0.0
        executed = self.metrics["passed"] + self.metrics["failed"] + self.metrics["blocked"]
        return (executed / self.metrics["total_test_cases"]) * 100
    
    def quality_index(self) -> float:
        """Индекс качества = Pass Rate * (1 - Defect Ratio)."""
        if self.metrics["total_test_cases"] == 0:
            return 0.0
        defect_ratio = self.metrics["failed"] / self.metrics["total_test_cases"]
        return self.pass_rate() * (1 - defect_ratio)
    
    def generate_report(self) -> str:
        return f"""
        === TESTING DASHBOARD ===
        Total Test Cases: {self.metrics['total_test_cases']}
        Passed: {self.metrics['passed']} ✅
        Failed: {self.metrics['failed']} ❌
        Blocked: {self.metrics['blocked']} 🔴
        Not Run: {self.metrics['not_run']} ⏸️
        
        Pass Rate: {self.pass_rate():.1f}%
        Execution Rate: {self.execution_rate():.1f}%
        Quality Index: {self.quality_index():.1f}%
        
        Open Bugs: {self.metrics['open_bugs']}
        Critical: {self.metrics['critical_bugs']}
        High: {self.metrics['high_bugs']}
        """
