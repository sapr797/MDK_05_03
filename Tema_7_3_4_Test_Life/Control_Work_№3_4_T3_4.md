Задание 1 (20 баллов). Комплексное тестирование с моками
Разработайте полный набор тестов (модульные + интеграционные) для сервиса заказов:

python
class OrderService:
    def __init__(self, payment_gateway, inventory, notifier):
        self.payment = payment_gateway
        self.inventory = inventory
        self.notifier = notifier
    
    def create_order(self, user_id: int, items: List[dict]) -> dict:
        # 1. Проверка наличия
        for item in items:
            if not self.inventory.check(item["product_id"], item["quantity"]):
                raise OutOfStockError(item["product_id"])
        
        # 2. Резервирование
        for item in items:
            self.inventory.reserve(item["product_id"], item["quantity"])
        
        # 3. Расчёт суммы
        total = sum(item["price"] * item["quantity"] for item in items)
        
        # 4. Платёж
        transaction = self.payment.charge(user_id, total)
        
        # 5. Уведомление
        self.notifier.send_order_confirmation(user_id, transaction["id"])
        
        return {"order_id": transaction["id"], "total": total}
