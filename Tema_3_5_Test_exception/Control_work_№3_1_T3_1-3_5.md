class ShoppingCart:
    def __init__(self):

        self._total = 0.0
    
    def add_item(self, name, price, quantity=1):
        if not name or not isinstance(name, str):
            raise ValueError("Product name must be non-empty string")
        if price <= 0:
            raise ValueError("Price must be positive")
        if quantity <= 0:
            raise ValueError("Quantity must be positive")
        
        self._items.append({"name": name, "price": price, "quantity": quantity})
        self._total += price * quantity
    
    def remove_item(self, index):
        if index < 0 or index >= len(self._items):
            raise IndexError("Invalid item index")
        item = self._items.pop(index)
        self._total -= item["price"] * item["quantity"]
    
    def get_total(self):
        return round(self._total, 2)
    
    def get_items(self):
        return self._items.copy()
    
    def clear(self):
        self._items.clear()
        self._total = 0.0
    
    def apply_discount(self, percent):
        if not 0 <= percent <= 100:
            raise ValueError("Discount must be between 0 and 100")
        self._total = self._total * (100 - percent) / 100
